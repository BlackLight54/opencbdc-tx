37auto decode_address(const std::string& addr_str)
38 -> std::optional<cbdc::hash_t> {
39 const auto [hrp, enc_data] = bech32::Decode(addr_str);
40 if(hrp != cbdc::config::bech32_hrp) {
41 throw "Invalid address encoding";
42 return std::nullopt;
43 }
44 auto data = std::vector<uint8_t>();
45 ConvertBits<bech32_bits_per_symbol, bits_per_byte, false>(
46 [&](uint8_t c) {
47 data.push_back(c);
48 },
49 enc_data.begin(),
50 enc_data.end());
51
52 auto pubkey = cbdc::hash_t();
53 if(data[0] != static_cast<uint8_t>(cbdc::client::address_type::public_key)
54 || data.size() != pubkey.size() + 1) {
55 throw "Address is not a supported type";
56 return std::nullopt;
57 }
58 data.erase(data.begin());
59 std::copy_n(data.begin(), pubkey.size(), pubkey.begin());
60
61 return pubkey;
62}
